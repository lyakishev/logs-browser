Программа для просмотра логов

Инcтрукция по использованию

Описание файла настроек logsbrowser.cfg

[db]
sql_uri = место хранения БД, возможные значения - :memory: - в памяти, пусто - в памяти с использованием диска при нехватке памяти, имя_файла - в файле

[config_files]
file_logs = конфигурационный файл источников файловых логов
event_logs = кнофигурационный файл источников евент-логов
syntax = конфигурационный файл с вариантами посветки в окне просмотра логов
queries = файл с сохраненными запросами, доступными из области редактирования запроса

[ui]
default_ftsindex = использование Full-Text-Index по-умолчанию

[colors]
blink_ms = не используется
bold_selected = метод выделения подсвеченных строк, False - как обычно, True - изменением шрифта в строке на полужирный.
color_rectangle_width = ширина параллелограмма одного цвета при подсветке строки в таблице несколькими цветами

[time]
syncronize = синхронизация времени с удаленным сервером
syncronize_server = сервер синхронизации времени
server_time_format = формат времени выдаваемый на сервере командой net time

[multiprocessing]
multiprocess = использовать нсеколько потоков или нет

[table]
grid_default = отображение сетки в таблице по-умолчанию

[tree]
fill_at_start = аполнять дерево источников файловых логов при запуске программы

[config_editor]
external_config_editor = внешний редактор конфигурационных файлов

[query]
default_where_operator = не используется
default_where_operator_fts = не используется

[logwindow]
external_log_viewer = в какой программе открывать файл лога при нажатии на соответствующую кнопку в окне просмотра лога
